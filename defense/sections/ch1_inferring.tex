%%%%%%%%%%%%%%%%%%%%% (1) OVERVIEW %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}{Inferring firms' emissions from transaction data}

    \begin{tcolorbox}[colback=green!5,colframe=green!40!black]
        \textbf{Question:} what can we learn about firms' emissions from \textbf{transaction data}?
    \end{tcolorbox}

    \begin{wideitemize}
        \item \textbf{\alertg{Motivation:}} registries \& surveys reach only \alertr{large installations}, in a handful of countries; \alerto{transaction data} covers \emph{every formal firm by design}
        \item \textbf{\alertb{Idea:}} combustion requires \alerto{buying fuel} $\rightarrow$ purchases leave traces in the \alertp{network}
        \item \textbf{\alertoch{Data:}} Belgium --- near-universe B2B $+$ \alertg{verified EU ETS emissions} as ground truth
        \item \textbf{\alertr{Result:}} a data-driven \alertp{elastic-net proxy} reliably tells \alertg{emitters from non-emitters} and ranks them by intensity --- beating size- and industry-code benchmarks
    \end{wideitemize}

\end{frame}

%%%%%%%%%%%%%%%%%%%%% (2) HOW WE BUILD THE PROXY %%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}{How we build the proxy}

    \begin{wideitemize}
        \item \textbf{\alertb{Train where emissions are observed}}:
        \begin{wideitemize}
            \item[-] \alertg{EU ETS firms} (verified) $+$ non-ETS firms in $\sim$100\%-covered energy sectors (assigned \alerto{zero})
            \item[-] the zeros teach the \alertb{extensive margin} --- and these energy-intensive sectors make it a demanding test
        \end{wideitemize}

        \item \textbf{\alertp{Let the data choose suppliers}}: regress emissions on \alertb{purchases from each supplier}, with an \alertp{elastic-net} penalty
        $$ \text{EN-proxy}_{it} \;=\!\! \sum_{j:\,|\hat\beta_j|>0}\!\! \hat\beta_j\, x_{ijt}
            \qquad {\scriptstyle x_{ijt}\,=\,\text{purchases of firm } i \text{ from supplier } j} $$

        \item \textbf{\alertoch{Benchmark}}: sum purchases from the six \alertr{fuel-selling NACE codes} (misses intermediaries; symmetric weights)
    \end{wideitemize}

\end{frame}

%%%%%%%%%%%%%%%%%%%%% (5) TAKEAWAYS %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}{Chapter 1 --- takeaways}

    \begin{wideitemize}
        \item \alertb{B2B transaction data carry a signal about emissions} beyond balance-sheet variables

        \item A \alertp{cross-fitted elastic-net proxy} beats industry-code classification; a hybrid captures both margins

        \item \alertg{Broader lesson}: combustion $\Rightarrow$ fuel purchases $\Rightarrow$ network traces --- a \alerto{general logic}, testable wherever transaction data $+$ a partial registry coexist

        \item[] {\color{dgray}\small $\hookrightarrow$ the same firm-level emissions \& network reappear, structurally, in Chapter~3.}
    \end{wideitemize}

\end{frame}
