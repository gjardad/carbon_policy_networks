%%%%%%%%%%%%%%%%%%%%% TEMPLATE-STYLE OVERVIEW %%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}{Carbon leakage: the problem in one slide}

    \begin{tcolorbox}[colback=red!4,colframe=red!50!black]
        \textbf{Question:} does the EU ETS push activity toward \emph{less-regulated} suppliers ---
        \alertr{within} Belgium (to less carbon-exposed domestic firms) and \alertr{across} its
        borders (to non-EU imports)?
    \end{tcolorbox}

    \begin{wideitemize}
        \item[\alertoch{Data:}] firm-level Belgium --- near-universe \alertb{B2B network} $+$ firm$\times$product$\times$country \alertp{customs} $+$ \alertg{EUTL} emissions/allowances
        \pause
        \item[\alertb{Approach:}] continuous-treatment DiD; dose $=$ supplier's predetermined carbon-cost exposure $\omega$; shock $=$ the \alerto{2015 MSR}-driven EUA run-up; follow the buyers with the strongest incentive to switch
        \pause
        \item[\alertr{Result:}] carbon pricing \alertg{does} raise producer prices (more in exposed sectors) --- but sourcing \alertr{does not move}. \textbf{Three margins, three nulls.}
        \pause
        \item[\alertp{Why:}] the relative-price gap stays \alertr{below 1\%} even at full pass-through --- too small to repay switching
    \end{wideitemize}

\end{frame}

%%%%%%%%%%%%%%%%%%%%% TAKEAWAYS %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}{Chapter 2 --- three margins, three nulls}

    \begin{tcolorbox}[colback=red!4,colframe=red!50!black]
        \centering The set of margins with detectable leakage is \textbf{empty}.
    \end{tcolorbox}

    \begin{wideitemize}
        \item[\alertg{1.}] Pass-through is \alertg{real} and persistent --- more in shortage-exposed sectors
        \item[\alertr{2.}] \alertr{No} within-NACE reallocation: wrong-signed, $\div 4$ and insignificant, not robust
        \item[\alertr{3.}] \alertr{No} international reallocation: flat shares; Belgium two orders of magnitude below France
    \end{wideitemize}

    \vspace{0.2cm}
    \alertp{Mechanism}: the cost shock is small and pass-through partial $\Rightarrow$ the relative-price gap never clears the cost of switching suppliers.

    \vspace{0.15cm}
    {\color{dgray}\small $\hookrightarrow$ Chapters 1--2 measure firms and test reallocation \emph{reduced-form};
    Chapter~3 builds the structural model that says \emph{when} reallocation should bind.}

\end{frame}
