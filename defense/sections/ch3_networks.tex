%%%%%%%%%%%%%%%%%%%%% (1) OVERVIEW %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}{Carbon policies in production networks}

    \begin{tcolorbox}[colback=blue!4,colframe=blue!40!black]
        \textbf{Question:} a carbon price covers \emph{some} firms, not all --- what is its
        effect on aggregate emissions, and how does it depend on \emph{whom} it covers?
    \end{tcolorbox}

    \begin{wideitemize}
        \item \textbf{\alertb{Theory:}} GE production network, heterogeneous emission intensities $+$ costly \alertp{abatement}; carbon price $p_z$, $0/1$ coverage $\tau$
        \item \textbf{\alertg{Result:}} $\tfrac{d\log Z}{dp_z} =$ \alerto{scale} ($\approx 0$) $+$ \alertp{technique} $+$ \alertb{reallocation} --- the last a \textbf{single covariance} of exposure with network-adjusted intensity $\Psi_e$
        \item \textbf{\alertoch{Data:}} Belgium (B2B $+$ EUTL): under the EU ETS the cut is \alertr{mostly abatement}, reallocation a modest amplifier
        \item \textbf{\alertr{Punchline:}} almost all the gain from \emph{whom to cover} is firms' \alertr{own} emissions --- a constrained regulator just \alertb{targets the heaviest emitters}
    \end{wideitemize}

\end{frame}

%%%%%%%%%%%%%%%%%%%%% (3) ENVIRONMENT + ABATEMENT %%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}{Environment: a production network with abatement}

    \begin{wideitemize}
        \item Closed economy, $N$ producers (CRS, marginal-cost pricing); \alertp{nested-CES} inputs ($\sigma_B$ across sectors, $\sigma_W$ within); Domar weights $\lambda_i$, Leontief $\Psi=(I-\Omega)^{-1}$

        \item \textbf{\alertb{Policy}}: carbon price $p_z$, coverage $\tau\in\{0,1\}^N$; emissions a by-product, intensity $\overline{e}_i$

        \item \textbf{\alertp{Covered firms abate}} (effort $\kappa_i$, $\alpha_i$ $=$ how hard $i$ is to abate):
        $$ e_i = (1-\kappa_i)^{\alpha_i}\,\overline{e}_i
            \qquad\Longrightarrow\qquad
            \frac{d \log e_i}{d p_z}\bigg|_{0} = -\,\tau_i\,\alpha_i^2\,\overline{e}_i \;<\;0 $$

        \item \textbf{\alertg{Key separability}}: abatement lowers intensity but leaves the input mix --- hence pass-through --- unchanged to first order
    \end{wideitemize}

\end{frame}

%%%%%%%%%%%%%%%%%%%%% (4) PRICE PROPAGATION %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}{The carbon price propagates through prices}

    \begin{wideitemize}
        \item Each firm passes its carbon cost into its price; buyers inherit it through inputs:
        $$ \frac{d \log p_i}{d p_z} = \sum_{j\in\N}\Omega_{ij}\,\frac{d \log p_j}{d p_z} \;+\; \Omega_{iL}\,\frac{d \log w}{d p_z} \;+\; \tau_i\,\overline{e}_i $$

        \item In matrix form --- the \alertb{exposure} of every price to the policy:
        $$ \frac{d \log \mathbf{p}}{d p_z} = \Psi\,(\tau \circ \overline{\mathcal{E}}) \;+\; \Psi_{(:,L)}\frac{d \log \lambda_L}{d p_z} $$

        \item[] \alertg{$\Rightarrow$} a firm's exposure is its \emph{covered}, network-weighted upstream emission content
    \end{wideitemize}

\end{frame}

%%%%%%%%%%%%%%%%%%%%% (5) MAIN RESULT / THEOREM %%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}{Main result: a sufficient-statistic decomposition}

    Three channels --- and \alerto{scale} $\approx 0$ by Hulten ($\tfrac{d\log Y}{dp_z}\big|_0=0$):

    \begin{tcolorbox}[colback=blue!4,colframe=blue!45!black]
    $$ \frac{d \log Z}{d p_z}\bigg|_{p_z=0}
        = \underbrace{-\sum_{i \in \N} \frac{\tau_i z_i}{Z}\,\alpha_i^2\,\overline{e}_i}_{\text{\alertp{technique} $<0$, covered firms}}
        \;\underbrace{-\,\sigma \sum_{i \in \N \cup \{C\}} \frac{x_i}{Z}\,\mathbb{C}_{\Omega_{(i,:)}}\!\left(\frac{d \log \mathbf{p}}{d p_z},\,\Psi_e\right)}_{\text{\alertb{reallocation}: one covariance}} $$
    \end{tcolorbox}

    \begin{wideitemize}
        \item Reallocation $=$ one covariance: \alertb{exposure} vs.\ \alertg{network-adjusted intensity} $\Psi_e$
        \item Its \alertr{sign} decides \alertg{amplify} ($\mathbb{C}>0$) vs.\ \alerto{leak} ($\mathbb{C}<0$); $(1-\sigma)$ sets the \alertp{magnitude}
        \item \alertr{Complete coverage}: covariance $\to$ \alertg{variance} $\Rightarrow$ reallocation \emph{always} amplifies
    \end{wideitemize}

\end{frame}

%%%%%%%%%%%%%%%%%%%%% (6) EMISSION CENTRALITY %%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}{Whom to cover? Emission centrality}

    \begin{wideitemize}
        \item The decomposition is \alertb{additively separable} across covered firms $\Rightarrow$ the marginal value of covering firm $j$ alone:
        $$ \text{Centrality}_j \;=\; \underbrace{-\frac{z_j}{Z}\,\alpha_j^2\,\overline{e}_j}_{\text{\alertp{own abatement}}}
            \;\underbrace{-\; \sigma \sum_{i \in \N \cup \{C\}} \frac{x_i}{Z}\,\mathbb{C}_{\Omega_{(i,:)}}\!\Big(\Psi_{(:,j)}\,\overline{e}_j,\ \Psi_e\Big)}_{\text{\alertb{network position}}} $$

        \item The object counterfactuals \alertg{rank firms on} --- and it makes the network's value an \emph{empirical} question
    \end{wideitemize}

\end{frame}

%%%%%%%%%%%%%%%%%%%%% (8) RESULT 1 %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}{Result 1 --- the EU ETS effect is mostly abatement}

    \begin{center}
    \begin{tabular}{lcc}
    \toprule
    Channel & $\Delta\log Z$ & share \\
    \midrule
    \textbf{Total}            & \alertb{$-0.421$} & \\
    \quad Scale               & $-0.000$ & $\approx 0$ \\
    \quad \alertp{Technique}  & \alertp{$-0.371$} & $\approx 88\%$ (\,$\sim$7/8\,) \\
    \quad \alertb{Composition}& \alertb{$-0.049$} & $\approx 12\%$ \\
    \bottomrule
    \end{tabular}
    \end{center}

    \begin{wideitemize}
        \item Composition is \alertg{negative} $\Rightarrow$ the Belgian network \alertg{amplifies} abatement (covered firms sit in cleaner-leaning supply chains)
        \item But \alerto{uneven}: taxing a firm lowers emissions at \alertb{$\sim$4 in 5} firms, raises them at the rest (offsetting \alertr{$\sim$1/5})
        \item For \alertr{$\sim$1 in 25} firms, coverage \emph{backfires}: demand shifts to dirtier uncovered rivals
    \end{wideitemize}

\end{frame}

%%%%%%%%%%%%%%%%%%%%% (10) TAKEAWAYS %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}{Chapter 3 --- takeaways}

    \begin{wideitemize}
        \item A \alertb{transparent decomposition} of targeted carbon pricing: scale ($\approx 0$), technique, and reallocation --- the last a \alertg{single covariance} of exposure with network-adjusted intensity $\Psi_e$

        \item Empirically (Belgium, EU ETS): the cut is \alertp{mostly abatement}; the network \alertg{amplifies} modestly

        \item The network is indispensable for \alertb{diagnosis} --- measuring the effect and flagging the firms whose coverage \alertr{backfires} --- but for \alerto{selection} a constrained regulator simply \alertr{targets the heaviest emitters}

        \item[] \alertp{Next}: open economy (cross-border leakage --- Chapter~2), and the jointly-optimal covered set
    \end{wideitemize}

\end{frame}
